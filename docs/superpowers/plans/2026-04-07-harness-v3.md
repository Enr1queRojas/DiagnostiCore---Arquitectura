**Goal:** Refactor DiagnostiCore into a Harness v3 system with explicit Shift-Worker phase functions, Context_Node handoffs, BuildRound tracking, deterministic verification matrix, and an `AgentRunner` class — fully backward-compatible with `main.py` and `api/app.py`.

**Architecture:** A new `Harness` coordinator in `orchestrator/harness.py` replaces the body of `_execute_pipeline()` with four explicit phase methods (`planning`, `dimensional_sprint`, `synthesis`, `output`). """
```python
orchestrator/handoff_exceptions.py
====================================
Exceptions raised by the Context_Node handoff and verification layers
of the Harness v3 architecture.
"""

from orchestrator.exceptions import OrchestratorError


class HandoffMismatchError(OrchestratorError):
    """Raised when a phase reads a handoff intended for a different phase."""

    def __init__(self, expected_phase: str, actual_next_phase: str):
        self.expected_phase = expected_phase
        self.actual_next_phase = actual_next_phase
        super().__init__(
            f"Handoff phase mismatch: phase '{expected_phase}' tried to read "
            f"a handoff whose stored next_phase is '{actual_next_phase}'."
        )


class ContextNodeIntegrityError(OrchestratorError):
    """Raised when a context_node payload checksum verification fails."""

    def __init__(self, expected: str, actual: str):
        self.expected = expected
        self.actual = actual
        super().__init__(
            f"Context node integrity failure: stored checksum '{expected}' "
            f"does not match recomputed checksum '{actual}'. The state file "
            f"may have been corrupted between sessions."
        )
```

```python
# ─────────────────────────────────────────────────────────────────────────────
# Context_Node handoff API (Harness v3)
# ─────────────────────────────────────────────────────────────────────────────

def _canonical_checksum(payload: dict) -> str:
    """Return a deterministic SHA-256 of a JSON-serialisable payload."""
    canonical = json.dumps(payload, sort_keys=True, ensure_ascii=False).encode("utf-8")
    return "sha256:" + hashlib.sha256(canonical).hexdigest()


def write_handoff(
    diagnostico_id: str,
    current_phase: str,
    next_phase: str,
    payload: dict,
    produced_by: str,
) -> None:
    """
    Persist the Minimal Viable Context handoff between two pipeline phases.

    The payload is opaque to state_manager — each phase serialises its own
    typed dataclass. A SHA-256 checksum is stored alongside it so the consumer
    can verify integrity on read.
    """
    lock_path = _STATE_FILE.with_suffix(".lock")
    fh = _acquire_lock(lock_path)
    try:
        state = _read_raw()
        _assert_id(state, diagnostico_id)
        state.setdefault("context_node", _blank_state(diagnostico_id)["context_node"])
        state["context_node"]["current_phase"] = current_phase
        state["context_node"]["next_phase"] = next_phase
        state["context_node"]["produced_by"] = produced_by
        state["context_node"]["produced_at"] = _now_iso()
        state["context_node"]["consumed_at"] = None
        state["context_node"]["payload"] = payload
        state["context_node"]["checksum"] = _canonical_checksum(payload)
        state["updated_at"] = _now_iso()
        _write_raw(state)
        logger.debug(
            "Handoff written | id=%s | %s -> %s | keys=%s",
            diagnostico_id, current_phase, next_phase, list(payload.keys()),
        )
    finally:
        _release_lock(fh)


def read_handoff(diagnostico_id: str, expected_phase: str) -> dict:
    """
    Read the current handoff and assert it was written for `expected_phase`.

    Verifies the checksum, marks the node as consumed, appends an entry to
    the rolling history (last 20), and returns a copy of the node dict.
    """
    # Lazy import to avoid circular dependency at module load time
    from orchestrator.handoff_exceptions import (
        HandoffMismatchError, ContextNodeIntegrityError,
    )

    lock_path = _STATE_FILE.with_suffix(".lock")
    fh = _acquire_lock(lock_path)
    try:
        state = _read_raw()
        _assert_id(state, diagnostico_id)
        node = state.get("context_node", {})
        actual_next = node.get("next_phase", "")
        if actual_next != expected_phase:
            raise HandoffMismatchError(expected_phase, actual_next)

        # Integrity check
        recomputed = _canonical_checksum(node.get("payload", {}))
        stored = node.get("checksum", "")
        if stored and stored != recomputed:
            raise ContextNodeIntegrityError(stored, recomputed)

        # Mark as consumed and rotate history
        node["consumed_at"] = _now_iso()
        history_entry = {
            "phase": node.get("produced_by", ""),
            "produced_at": node.get("produced_at", ""),
            "consumed_at": node["consumed_at"],
            "payload_keys": sorted(list(node.get("payload", {}).keys())),
        }
        history = node.setdefault("history", [])
        history.append(history_entry)
        if len(history) > _HANDOFF_HISTORY_LIMIT:
            del history[: len(history) - _HANDOFF_HISTORY_LIMIT]

        state["context_node"] = node
        state["updated_at"] = _now_iso()
        _write_raw(state)
        return dict(node)
    finally:
        _release_lock(fh)


def clear_handoff(diagnostico_id: str) -> None:
    """Reset the active handoff (history is preserved)."""
    lock_path = _STATE_FILE.with_suffix(".lock")
    fh = _acquire_lock(lock_path)
    try:
        state = _read_raw()
        _assert_id(state, diagnostico_id)
        node = state.get("context_node", {})
        history = node.get("history", [])
        state["context_node"] = {
            "current_phase": "",
            "next_phase": "",
            "produced_by": "",
            "produced_at": "",
            "consumed_at": None,
            "payload": {},
            "checksum": "",
            "history": history,
        }
        state["updated_at"] = _now_iso()
        _write_raw(state)
    finally:
        _release_lock(fh)
```

```python
# ─────────────────────────────────────────────────────────────────────────────
# BuildRound API (Harness v3)
# ─────────────────────────────────────────────────────────────────────────────

VALID_ROUND_OUTCOMES = {"running", "generated", "pass", "fail", "escalated"}
VALID_ROUND_TRIGGERS = {
    "initial_run", "after_round_1", "after_round_2",
    "qa_feedback_round_1", "qa_feedback_round_2",
    "manual_retry", "escalation_recovery",
}


def start_round(
    diagnostico_id: str,
    dim_key: str,
    agent: str,
    trigger: str,
) -> str:
    """
    Begin a new round for a dimension. Returns the round_id.

    Naming convention:
        round_1, round_1_qa, round_2, round_2_qa, ...
    """
    if trigger not in VALID_ROUND_TRIGGERS:
        raise ValueError(
            f"Invalid trigger {trigger!r}. Valid: {VALID_ROUND_TRIGGERS}"
        )

    lock_path = _STATE_FILE.with_suffix(".lock")
    fh = _acquire_lock(lock_path)
    try:
        state = _read_raw()
        _assert_id(state, diagnostico_id)
        state.setdefault("build_rounds", {})
        dim_entry = state["build_rounds"].setdefault(
            dim_key, {"current_round": 0, "rounds": []}
        )

        # Determine the next round_id based on the last round name
        existing = [r["round_id"] for r in dim_entry["rounds"]]
        if not existing:
            round_id = "round_1"
            dim_entry["current_round"] = 1
        else:
            last = existing[-1]
            if last.endswith("_qa"):
                # Last was a QA round, next is the next generation round
                n = int(last.replace("round_", "").replace("_qa", ""))
                round_id = f"round_{n + 1}"
                dim_entry["current_round"] = n + 1
            else:
                # Last was a generation round, next is its QA pair
                round_id = f"{last}_qa"

        dim_entry["rounds"].append({
            "round_id": round_id,
            "agent": agent,
            "started_at": _now_iso(),
            "finished_at": None,
            "outcome": "running",
            "output_path": None,
            "eval_path": None,
            "score": None,
            "feedback_summary": None,
            "trigger": trigger,
        })
        state["updated_at"] = _now_iso()
        _write_raw(state)
        logger.debug("Round started | id=%s | dim=%s | round=%s", diagnostico_id, dim_key, round_id)
        return round_id
    finally:
        _release_lock(fh)


def complete_round(
    diagnostico_id: str,
    dim_key: str,
    round_id: str,
    outcome: str,
    output_path: str | None = None,
    eval_path: str | None = None,
    score: float | None = None,
    feedback_summary: str | None = None,
) -> None:
    """Mark an existing round as finished with the given outcome."""
    if outcome not in VALID_ROUND_OUTCOMES:
        raise ValueError(
            f"Invalid outcome {outcome!r}. Valid: {VALID_ROUND_OUTCOMES}"
        )

    lock_path = _STATE_FILE.with_suffix(".lock")
    fh = _acquire_lock(lock_path)
    try:
        state = _read_raw()
        _assert_id(state, diagnostico_id)
        dim_entry = state.get("build_rounds", {}).get(dim_key)
        if not dim_entry:
            raise KeyError(f"No rounds started for dimension {dim_key!r}")

        for r in dim_entry["rounds"]:
            if r["round_id"] == round_id:
                r["finished_at"] = _now_iso()
                r["outcome"] = outcome
                if output_path is not None:
                    r["output_path"] = output_path
                if eval_path is not None:
                    r["eval_path"] = eval_path
                if score is not None:
                    r["score"] = score
                if feedback_summary is not None:
                    r["feedback_summary"] = feedback_summary[:200]
                break
        else:
            raise KeyError(f"Round {round_id!r} not found for {dim_key!r}")

        state["updated_at"] = _now_iso()
        _write_raw(state)
    finally:
        _release_lock(fh)


def get_round_history(diagnostico_id: str, dim_key: str) -> list[dict]:
    """Return the list of rounds for a dimension (empty list if none)."""
    state = load_state(diagnostico_id)
    return state.get("build_rounds", {}).get(dim_key, {}).get("rounds", [])


def get_current_round(diagnostico_id: str, dim_key: str) -> int:
    """Return the current round number (0 if none)."""
    state = load_state(diagnostico_id)
    return state.get("build_rounds", {}).get(dim_key, {}).get("current_round", 0)
```

```python
    """
    Load the current state for the given diagnostico_id. Auto-upgrades
    schema_version 2.0 files in-memory by injecting empty context_node and
    build_rounds blocks. The upgrade is persisted on the next write.

    Raises:
        FileNotFoundError: If no state file exists yet.
        ValueError: If the stored state belongs to a different diagnostico_id.
    """
    if not _STATE_FILE.exists():
        raise FileNotFoundError(
            f"No state file found at {_STATE_FILE}. "
            "Call init_state() to create a new diagnostic."
        )
    state = _read_raw()
    stored_id = state.get("diagnostico_id", "")
    if stored_id and stored_id != diagnostico_id:
        raise ValueError(
            f"State file contains diagnostico_id='{stored_id}', "
            f"but requested '{diagnostico_id}'. "
            "Call init_state() to start a new diagnostic."
        )

    # ── Auto-upgrade 2.0 -> 3.0 ──────────────────────────────────────────────
    meta = state.setdefault("_meta", {})
    if meta.get("schema_version") in ("2.0", None):
        meta["schema_version"] = "3.0"
        state.setdefault("context_node", {
            "current_phase": "", "next_phase": "", "produced_by": "",
            "produced_at": "", "consumed_at": None, "payload": {},
            "checksum": "", "history": [],
        })
        state.setdefault("build_rounds", {})
        logger.info("State auto-upgraded 2.0 -> 3.0 | id=%s", diagnostico_id)

    return state
```

```python
orchestrator/build_rounds.py
==============================
Helper utilities for the BuildRound tracking system.

Build Rounds capture the audit trail of dimensional retries: each generation

The persistence layer lives in state_manager — this module provides typed

class RoundTrigger(str, Enum):
    """Why a new round was started. Mirrors state_manager.VALID_ROUND_TRIGGERS."""
    INITIAL_RUN = "initial_run"
    AFTER_ROUND_1 = "after_round_1"
    AFTER_ROUND_2 = "after_round_2"
    QA_FEEDBACK_ROUND_1 = "qa_feedback_round_1"
    QA_FEEDBACK_ROUND_2 = "qa_feedback_round_2"
    MANUAL_RETRY = "manual_retry"
    ESCALATION_RECOVERY = "escalation_recovery"


class RoundOutcome(str, Enum):
    """Possible terminal states of a round. Mirrors VALID_ROUND_OUTCOMES."""
    RUNNING = "running"
    GENERATED = "generated"
    PASS = "pass"
    FAIL = "fail"
    ESCALATED = "escalated"


def diff_rounds(rounds: list[dict]) -> str:
    """
    Produce a single-line human-readable diff between successive rounds.
    Used by the harness for log lines and history.
    """
    if not rounds:
        return "(no rounds)"
    parts = []

        parts.append(f"{r['round_id']}({r['outcome']}{score_str})")
    return " -> ".join(parts)
```

```python
"""
orchestrator/verification_matrix.py
=====================================
Deterministic, rule-based pre-validator for agent outputs.

Acts as the 'Human Surrogate' / 'Compensatory Scaffold' layer from the
Autonomous Agent Blueprint: catches structural failures cheaply BEFORE the
LLM-based A9/A10 evaluators run, eliminating obvious failure classes
deterministically and saving API spend.

Rules are loaded from config/acceptance_criteria.json and dispatched by name
to pure Python functions defined in this module. Each rule receives the agent
output dict and a parameters dict, and returns True (pass) or False (fail).
"""

from __future__ import annotations

import json
import logging
import re
from dataclasses import dataclass, field
from enum import Enum
from pathlib import Path

logger = logging.getLogger(__name__)

_ROOT = Path(__file__).parent.parent
_CONFIG_DIR = _ROOT / "config"


class Severity(str, Enum):
    """Severity ladder for matrix criteria."""
    CRITICAL = "critical"
    WARNING = "warning"


@dataclass
class MatrixResult:
    """The deterministic verdict for one agent output."""
    passed: bool
    failures: list[dict] = field(default_factory=list)
    warnings: list[dict] = field(default_factory=list)
    feedback_for_agent: str = ""


# ─────────────────────────────────────────────────────────────────────────────
# Rule functions (pure)
# ─────────────────────────────────────────────────────────────────────────────

def _rule_list_length(output: dict, params: dict) -> bool:
    """Check that output[field] is a list of length within [min, max]."""
    field_name = params["field"]
    val = output.get(field_name)
    if not isinstance(val, list):
        return False
    n = len(val)
    return params.get("min", 0) <= n <= params.get("max", float("inf"))


def _rule_numeric_range(output: dict, params: dict) -> bool:
    """Check that output[field] is a number in [min, max]."""
    val = output.get(params["field"])
    if not isinstance(val, (int, float)):
        return False
    return params.get("min", float("-inf")) <= val <= params.get("max", float("inf"))


def _rule_min_length(output: dict, params: dict) -> bool:
    """Check that output[field] is a string with at least min characters."""
    val = output.get(params["field"])
    if not isinstance(val, str):
        return False
    return len(val) >= params.get("min", 0)


def _rule_min_sources(output: dict, params: dict) -> bool:
    """Heuristic: count distinct evidence sources cited in justificacion."""
    val = output.get("justificacion", "")
    if not isinstance(val, str):
        return False
    # Split on punctuation likely to separate citations and count unique tokens
    tokens = re.split(r"[,;|]\s*|\s+y\s+", val.lower())
    distinct = {t.strip() for t in tokens if len(t.strip()) > 4}
    return len(distinct) >= params.get("min_sources", 2)


def _rule_no_forbidden_terms(output: dict, params: dict) -> bool:
    """Return True if none of the forbidden terms appear in output[field]."""
    val = output.get(params["field"], "")
    if not isinstance(val, str):
        return True
    text = val.lower()
    return not any(term.lower() in text for term in params.get("forbidden", []))


def _rule_nested_list_length(output: dict, params: dict) -> bool:
    """
    Check a JSONPath-like expression `parent[*].child` such that every
    `child` list has length >= min.
    """
    path = params["path"]
    m = re.match(r"^(\w+)\[\*\]\.(\w+)$", path)
    if not m:
        return False
    parent_key, child_key = m.group(1), m.group(2)
    parent_list = output.get(parent_key, [])
    if not isinstance(parent_list, list):
        return False
    min_required = params.get("min", 0)
    for item in parent_list:
        child = item.get(child_key, [])
        if not isinstance(child, list) or len(child) < min_required:
            return False
    return True


def _rule_noop(output: dict, params: dict) -> bool:
    """Always passes. Used for criteria that are evaluated only by the LLM."""
    return True


_RULES = {
    "field.list_length": _rule_list_length,
    "field.numeric_range": _rule_numeric_range,
    "field.min_length": _rule_min_length,
    "justificacion.min_sources": _rule_min_sources,
    "text.no_forbidden_terms": _rule_no_forbidden_terms,
    "nested.list_length": _rule_nested_list_length,
    "noop": _rule_noop,
}


# ─────────────────────────────────────────────────────────────────────────────
# Criteria loader (cached)
# ─────────────────────────────────────────────────────────────────────────────

_CRITERIA_CACHE: dict | None = None


def _load_all_criteria() -> dict:
    """Load and cache the per-agent criteria map from acceptance_criteria.json."""
    global _CRITERIA_CACHE
    if _CRITERIA_CACHE is not None:
        return _CRITERIA_CACHE
    path = _CONFIG_DIR / "acceptance_criteria.json"
    if not path.exists():
        logger.warning("acceptance_criteria.json not found — verification matrix will pass everything")
        _CRITERIA_CACHE = {}
        return _CRITERIA_CACHE
    with open(path, "r", encoding="utf-8") as fh:
        data = json.load(fh)
    # Filter out the _meta block; keep only per-agent keys
    _CRITERIA_CACHE = {k: v for k, v in data.items() if k != "_meta" and isinstance(v, list)}
    return _CRITERIA_CACHE


def _criteria_for(agent_id: str) -> list[dict]:
    return _load_all_criteria().get(agent_id, [])


# ─────────────────────────────────────────────────────────────────────────────
# Public API
# ─────────────────────────────────────────────────────────────────────────────

def verify(agent_id: str, output: dict) -> MatrixResult:
    """
    Run all matrix rules registered for `agent_id` against `output`.

    Returns a MatrixResult. If `failures` is non-empty, the agent must
    regenerate using `feedback_for_agent`. Warnings are non-blocking and
    can be forwarded to A9/A10 as additional input.
    """
    criteria = _criteria_for(agent_id)
    failures: list[dict] = []
    warnings: list[dict] = []

    for criterion in criteria:
        rule_name = criterion.get("rule")
        rule_fn = _RULES.get(rule_name)
        if rule_fn is None:
            logger.warning("Unknown rule %r in criterion %s — skipping", rule_name, criterion.get("id"))
            continue
        try:
            ok = rule_fn(output, criterion.get("params", {}))
        except Exception as exc:
            logger.warning("Rule %s raised %s — counted as failure", rule_name, exc)
            ok = False
        if not ok:
            entry = {
                "id": criterion.get("id"),
                "rule": rule_name,
                "description": criterion.get("description", ""),
            }
            if criterion.get("severity") == Severity.CRITICAL.value:
                failures.append(entry)
            else:
                warnings.append(entry)

    feedback = _format_feedback(agent_id, failures, warnings)
    return MatrixResult(
        passed=not failures,
        failures=failures,
        warnings=warnings,
        feedback_for_agent=feedback,
    )


def _format_feedback(agent_id: str, failures: list[dict], warnings: list[dict]) -> str:
    """Build a concise human-readable feedback string for the agent retry."""
    if not failures and not warnings:
        return ""
    lines = [f"Verification matrix feedback for {agent_id}:"]
    if failures:
        lines.append("CRITICAL — must fix before retry:")
        for f in failures:
            lines.append(f"  - [{f['id']}] {f['description']}")
    if warnings:
        lines.append("WARNINGS — improve if possible:")
        for w in warnings:
            lines.append(f"  - [{w['id']}] {w['description']}")
    return "\n".join(lines)


def reset_cache() -> None:
    """For tests: forget the cached criteria so a new file is reloaded."""
    global _CRITERIA_CACHE
    _CRITERIA_CACHE = None
```

```python
"""
orchestrator/agent_runner_v2.py
=================================
AgentRunner — the per-agent Shift-Worker.

This class implements the Harness v3 lifecycle for one agent execution as
five explicit, individually-testable methods:

    1. load_context     — build the Minimal Viable Context for the agent
    2. execute_task     — single LLM call (no parsing, no validation)
    3. validate_output  — parse + JSON Schema + business rules + 1 self-correction
    4. structure_handoff— persist validated output via the blackboard write_lock
    5. release_context  — drop in-memory references (extension hook for future
                          context-window compaction strategies)

The legacy procedural run_agent() in agent_runner.py becomes a thin shim
around AgentRunner.execute() so all existing callers (api/app.py, main.py,
tests) continue to work unchanged.
"""

from __future__ import annotations

import logging
from dataclasses import dataclass, field
from typing import Any

from blackboard.blackboard import Blackboard
from orchestrator.exceptions import (
    AgentOutputError, OrchestratorError, ValidationError,
)
from orchestrator.llm_client import AsyncLLMClient

logger = logging.getLogger(__name__)


# ─────────────────────────────────────────────────────────────────────────────
# AgentContext — the Minimal Viable Context for one agent
# ─────────────────────────────────────────────────────────────────────────────

@dataclass
class AgentContext:
    """
    Frozen-by-convention dataclass holding everything an agent needs to run.

    Created by load_context() and passed through the rest of the lifecycle.
    Released at the end of execute() — held only for the duration of one run.
    """
    agent_id: str
    run_id: str
    system_prompt: str
    user_message: str
    raw_blackboard_context: dict = field(default_factory=dict)


# ─────────────────────────────────────────────────────────────────────────────
# AgentRunner — the Shift-Worker
# ─────────────────────────────────────────────────────────────────────────────

class AgentRunner:
    """Executes one agent through the 5-step Shift-Worker lifecycle."""

    def __init__(self, blackboard: Blackboard, llm_client: AsyncLLMClient):
        self.blackboard = blackboard
        self.llm_client = llm_client

    async def execute(self, agent_id: str, run_id: str) -> dict[str, Any]:
        """
        Top-level lifecycle. Calls the 5 methods in order and returns the
        validated output dict (also persisted to the blackboard).
        """
        ctx = self.load_context(agent_id, run_id)
        raw = await self.execute_task(agent_id, ctx)
        validated = await self.validate_output(agent_id, raw, ctx)
        await self.structure_handoff(agent_id, validated)
        self.release_context(ctx)
        return validated

    # ── 1. Load Context ──────────────────────────────────────────────────────
    def load_context(self, agent_id: str, run_id: str) -> AgentContext:
        """Build the Minimal Viable Context for the agent."""
        # Delegate to the existing helpers in agent_runner.py to keep
        # the prompt-injection guard and PII filter behaviour identical.
        from orchestrator.agent_runner import _build_user_message

        system_prompt = self.llm_client.load_system_prompt(agent_id)
        bb_ctx = self.blackboard.exportar_para_agente(agent_id)
        user_message = _build_user_message(bb_ctx)
        return AgentContext(
            agent_id=agent_id,
            run_id=run_id,
            system_prompt=system_prompt,
            user_message=user_message,
            raw_blackboard_context=bb_ctx,
        )

    # ── 2. Execute Task ──────────────────────────────────────────────────────
    async def execute_task(self, agent_id: str, ctx: AgentContext) -> str:
        """Single LLM call. Returns the raw response string."""
        messages = [{"role": "user", "content": ctx.user_message}]
        return await self.llm_client.sample(
            agent_id=agent_id,
            system_prompt=ctx.system_prompt,
            messages=messages,
        )

    # ── 3. Validate Output ───────────────────────────────────────────────────
    async def validate_output(
        self,
        agent_id: str,
        raw: str,
        ctx: AgentContext,
    ) -> dict[str, Any]:
        """Parse + JSON Schema validation + 1 self-correction round."""
        from orchestrator.agent_runner import _parse_validate_with_correction

        messages = [{"role": "user", "content": ctx.user_message}]
        return await _parse_validate_with_correction(
            agent_id=agent_id,
            raw_response=raw,
            messages=messages,
            system_prompt=ctx.system_prompt,
            llm_client=self.llm_client,
        )

    # ── 4. Structure Handoff ─────────────────────────────────────────────────
    async def structure_handoff(self, agent_id: str, validated: dict[str, Any]) -> None:
        """Persist the validated output to the blackboard."""
        from orchestrator.agent_runner import _write_to_blackboard
        await _write_to_blackboard(agent_id, self.blackboard, validated)

    # ── 5. Release Context ───────────────────────────────────────────────────
    def release_context(self, ctx: AgentContext) -> None:
        """
        Drop the in-memory context. No-op today, but this hook is in place so
        future versions can recycle context windows or trigger compaction
        without changing the lifecycle contract.
        """
        # Explicitly clear the largest fields so the GC can reclaim them sooner.
        ctx.raw_blackboard_context = {}
        ctx.user_message = ""
```

```python
"""
orchestrator/harness.py
=========================
The DiagnostiCore Harness — a Shift-Worker orchestrator that breaks the
diagnostic pipeline into four discrete phases, each with an explicit
5-step lifecycle:

    1. LOAD STATE       — read_handoff(expected_phase=...)
    2. EXECUTE          — do the work (contract / agents / synthesis / output)
    3. VALIDATE         — verification matrix + quality gate / onepager evaluator
    4. WRITE HANDOFF    — write_handoff(payload=asdict(TypedHandoff))
    5. RELEASE CONTEXT  — update_pipeline_status(); clear ephemeral state

Each phase function is independently testable and the pipeline can be resumed
across sessions via the context_node block in diagnostico-state.json.
"""

from __future__ import annotations

import asyncio
import logging
from dataclasses import asdict, dataclass, field
from typing import Any

from blackboard.blackboard import Blackboard
from orchestrator import state_manager
from orchestrator.agent_runner_v2 import AgentRunner
from orchestrator.build_rounds import RoundOutcome, RoundTrigger
from orchestrator.contract_builder import build_contract, load_contract
from orchestrator.exceptions import OrchestratorError
from orchestrator.llm_client import AsyncLLMClient
from orchestrator.onepager_evaluator import (
    OnePagerEscalationError, run_onepager_evaluation,
)
from orchestrator.quality_gate import (
    QualityGateEscalationError, run_quality_gate,
)

logger = logging.getLogger(__name__)


# ─────────────────────────────────────────────────────────────────────────────
# Typed handoff payloads — one dataclass per phase boundary
# ─────────────────────────────────────────────────────────────────────────────

@dataclass
class PlanningHandoff:
    """planning -> dimensional_sprint."""
    contract_path: str
    evidence_keys: list[str]
    client_summary: dict
    dimensions_to_run: list[str]


@dataclass
class DimensionalHandoff:
    """dimensional_sprint -> synthesis."""
    output_paths: dict[str, str]
    failed_dimensions: list[str]
    rounds_used: dict[str, int] = field(default_factory=dict)


@dataclass
class SynthesisHandoff:
    """synthesis -> output."""
    sintesis_path: str
    idd: float
    causas_raiz_count: int


@dataclass
class OutputHandoff:
    """output -> (terminal)."""
    onepager_path: str
    eval_passed: bool
    delivered: bool


# Static map: agent_id -> blackboard dimension key
_AGENT_DIMENSION_MAP: dict[str, str] = {
    "A1": "A1_estrategia",
    "A2": "A2_liderazgo",
    "A3": "A3_cultura",
    "A4": "A4_procesos",
    "A5": "A5_datos",
    "A6": "A6_tecnologia",
}


# ─────────────────────────────────────────────────────────────────────────────
# Harness coordinator
# ─────────────────────────────────────────────────────────────────────────────

class Harness:
    """
    Orchestrates the four phases of a DiagnostiCore run.
    Each phase reads its handoff from state, executes its work, writes the
    next handoff, and clears ephemeral context before returning.
    """

    def __init__(
        self,
        run_id: str,
        blackboard: Blackboard,
        llm_client: AsyncLLMClient,
        track_clean_runs: bool = False,
    ):
        self.run_id = run_id
        self.blackboard = blackboard
        self.llm_client = llm_client
        self.track_clean_runs = track_clean_runs
        self.runner = AgentRunner(blackboard, llm_client)
        self.results: dict[str, dict] = {}

    async def run(self) -> dict[str, dict]:
        """Execute the full pipeline. Each phase is a separate session-safe step."""
        await self._ensure_session_state()
        await self._phase_planning()
        await self._phase_dimensional_sprint()
        await self._phase_synthesis()
        await self._phase_output()
        return self.results

    # ── Session bootstrap ────────────────────────────────────────────────────
    async def _ensure_session_state(self) -> None:
        """Load or initialise diagnostico-state.json for this run."""
        try:
            s = state_manager.load_state(self.run_id)
            logger.info("Resuming diagnostic | id=%s | status=%s", self.run_id, s.get("status"))
        except FileNotFoundError:
            client_info = {}
            if hasattr(self.blackboard, "_data"):
                c = self.blackboard._data.get("cliente", {})
                client_info = {
                    "name": c.get("nombre", ""),
                    "industry": c.get("sector", ""),
                    "size": c.get("tamaño", ""),
                }
            state_manager.init_state(self.run_id, client_info)
            logger.info("Session state initialised | id=%s", self.run_id)

    # ── PHASE 1: Planning (contract building) ────────────────────────────────
    async def _phase_planning(self) -> None:
        """Generate the diagnostic contract before launching dimensional agents."""
        logger.info("═══ PHASE 1: Planning | run=%s ═══", self.run_id)

        # 1. LOAD STATE — nothing to read for the very first phase
        existing_contract = load_contract(self.run_id)

        # 2. EXECUTE
        client_info, evidence_keys = self._collect_planning_inputs()
        if not existing_contract:
            try:
                await build_contract(
                    diagnostico_id=self.run_id,
                    client_info=client_info,
                    available_evidence=evidence_keys,
                    llm_client=self.llm_client,
                )
            except Exception as exc:
                logger.warning(
                    "Contract generation failed (%s) — proceeding without contract", exc
                )

        # 3. VALIDATE — contract presence is enough; A9 will handle the rest
        contract_path = f"blackboard/contracts/{self.run_id}_contract.json"

        # 4. WRITE HANDOFF
        payload = PlanningHandoff(
            contract_path=contract_path,
            evidence_keys=evidence_keys,
            client_summary=client_info,
            dimensions_to_run=list(_AGENT_DIMENSION_MAP.values()),
        )
        state_manager.write_handoff(
            self.run_id,
            current_phase="planning",
            next_phase="dimensional_sprint",
            payload=asdict(payload),
            produced_by="phase_planning",
        )

        # 5. RELEASE CONTEXT
        state_manager.update_pipeline_status(self.run_id, "dimensions_running")

    def _collect_planning_inputs(self) -> tuple[dict, list[str]]:
        """Extract client info and available evidence keys from the blackboard."""
        client_info: dict = {}
        evidence_keys: list[str] = []
        if hasattr(self.blackboard, "_data"):
            c = self.blackboard._data.get("cliente", {})
            client_info = {
                "name": c.get("nombre", ""),
                "industry": c.get("sector", ""),
                "size": c.get("tamaño", ""),
            }
            ev = self.blackboard._data.get("evidencia", {})
            for key in ("transcripciones", "cuestionarios", "auditoria_tecnica",
                        "mapas_proceso", "series_financieras"):
                if ev.get(key):
                    evidence_keys.append(key)
        if not evidence_keys:
            evidence_keys = ["transcripciones"]
        return client_info, evidence_keys

    # ── PHASE 2: Dimensional Sprint ──────────────────────────────────────────
    async def _phase_dimensional_sprint(self) -> None:
        """Run A1-A6 in parallel + quality gate per dimension."""
        logger.info("═══ PHASE 2: Dimensional Sprint | run=%s ═══", self.run_id)

        # 1. LOAD STATE
        node = state_manager.read_handoff(self.run_id, expected_phase="dimensional_sprint")
        plan = PlanningHandoff(**node["payload"])
        self.blackboard.set_estado("analisis_dimensional")

        # 2. EXECUTE (parallel)
        dimensional_agents = ["A1", "A2", "A3", "A4", "A5", "A6"]
        outcomes = await asyncio.gather(
            *[self.runner.execute(aid, self.run_id) for aid in dimensional_agents],
            return_exceptions=True,
        )

        # 3. VALIDATE (per-dimension verification matrix + A9 quality gate)
        failed_dims: list[str] = []
        output_paths: dict[str, str] = {}
        rounds_used: dict[str, int] = {}

        for agent_id, outcome in zip(dimensional_agents, outcomes):
            dim_key = _AGENT_DIMENSION_MAP[agent_id]
            if isinstance(outcome, BaseException):
                logger.error("[%s] FAILED: %s", agent_id, outcome)
                self.blackboard.registrar_error(agent_id, str(outcome))
                failed_dims.append(dim_key)
                state_manager.update_dimension(self.run_id, dim_key, {"status": "failed"})
                state_manager.append_history(self.run_id, agent_id, "failed", str(outcome)[:200])
                continue

            self.results[agent_id] = outcome
            nivel = outcome.get("nivel_madurez", "?")
            output_path = f"blackboard/outputs/{self.run_id}_{agent_id}.json"
            state_manager.update_dimension(
                self.run_id, dim_key,
                {"status": "complete", "score": nivel, "output_path": output_path},
            )
            state_manager.append_history(self.run_id, agent_id, "complete", f"nivel={nivel}")
            output_paths[dim_key] = output_path

            # Quality gate (with retries)
            await self._gate_dimension(agent_id, dim_key, outcome, rounds_used)

        if failed_dims:
            raise OrchestratorError(
                f"Phase dimensional_sprint incomplete. Failed: {failed_dims}",
                failed_agents=[d.split("_")[0] for d in failed_dims],
            )

        # 4. WRITE HANDOFF
        payload = DimensionalHandoff(
            output_paths=output_paths,
            failed_dimensions=failed_dims,
            rounds_used=rounds_used,
        )
        state_manager.write_handoff(
            self.run_id,
            current_phase="dimensional_sprint",
            next_phase="synthesis",
            payload=asdict(payload),
            produced_by="phase_dimensional_sprint",
        )

        # 5. RELEASE CONTEXT
        state_manager.update_pipeline_status(self.run_id, "dimensions_complete")
        logger.info("Phase dimensional_sprint complete | rounds_used=%s", rounds_used)

    async def _gate_dimension(
        self,
        agent_id: str,
        dim_key: str,
        output: dict,
        rounds_used: dict[str, int],
    ) -> None:
        """Run A9 quality gate for one dimension, with retries on failure."""
        # Track the initial round (only persisted if a retry happens)
        round_id = state_manager.start_round(
            self.run_id, dim_key, agent_id, RoundTrigger.INITIAL_RUN.value,
        )
        state_manager.complete_round(
            self.run_id, dim_key, round_id, RoundOutcome.GENERATED.value,
            output_path=f"blackboard/outputs/{self.run_id}_{agent_id}.json",
        )

        passed, feedback = await run_quality_gate(
            diagnostico_id=self.run_id,
            dimension_key=dim_key,
            dimensional_output=output,
            llm_client=self.llm_client,
        )
        if passed:
            return

        # Retry with feedback
        rounds_used[dim_key] = 2
        logger.info("[QG] Retrying %s with quality-gate feedback...", agent_id)
        self.blackboard._data.setdefault("quality_gate_feedback", {})[agent_id] = feedback

        retry_round = state_manager.start_round(
            self.run_id, dim_key, agent_id, RoundTrigger.QA_FEEDBACK_ROUND_1.value,
        )
        try:
            self.results[agent_id] = await self.runner.execute(agent_id, self.run_id)
            state_manager.complete_round(
                self.run_id, dim_key, retry_round, RoundOutcome.GENERATED.value,
            )
        except Exception as retry_exc:
            state_manager.complete_round(
                self.run_id, dim_key, retry_round, RoundOutcome.FAIL.value,
                feedback_summary=str(retry_exc)[:200],
            )
            raise OrchestratorError(
                f"[{agent_id}] Failed on quality-gate retry: {retry_exc}",
                failed_agents=[agent_id],
            ) from retry_exc

        # Second QG pass
        passed_2, _ = await run_quality_gate(
            diagnostico_id=self.run_id,
            dimension_key=dim_key,
            dimensional_output=self.results[agent_id],
            llm_client=self.llm_client,
        )
        if not passed_2:
            # Will raise QualityGateEscalationError on the next call (retry_count >= 2)
            await run_quality_gate(
                diagnostico_id=self.run_id,
                dimension_key=dim_key,
                dimensional_output=self.results[agent_id],
                llm_client=self.llm_client,
            )

    # ── PHASE 3: Synthesis ───────────────────────────────────────────────────
    async def _phase_synthesis(self) -> None:
        """Run A7 synthesis on the validated dimensional outputs."""
        logger.info("═══ PHASE 3: Synthesis | run=%s ═══", self.run_id)

        # 1. LOAD STATE
        node = state_manager.read_handoff(self.run_id, expected_phase="synthesis")
        dim_h = DimensionalHandoff(**node["payload"])
        state_manager.update_pipeline_status(self.run_id, "synthesis")
        self.blackboard.set_estado("sintesis")

        # 2. EXECUTE
        a7_output = await self.runner.execute("A7", self.run_id)
        self.results["A7"] = a7_output

        # 3. VALIDATE — A7's schema is enforced inside AgentRunner.validate_output;
        # nothing extra here. Future: add an A7-specific quality gate.

        # 4. WRITE HANDOFF
        sintesis_path = f"blackboard/synthesis/{self.run_id}_A7.json"
        state_manager.update_synthesis(
            self.run_id, {"status": "complete", "output_path": sintesis_path},
        )
        payload = SynthesisHandoff(
            sintesis_path=sintesis_path,
            idd=float(a7_output.get("idd", 0.0)),
            causas_raiz_count=len(a7_output.get("causas_raiz", [])),
        )
        state_manager.write_handoff(
            self.run_id,
            current_phase="synthesis",
            next_phase="output",
            payload=asdict(payload),
            produced_by="phase_synthesis",
        )

        # 5. RELEASE CONTEXT
        state_manager.append_history(self.run_id, "A7", "complete",
                                     f"idd={payload.idd} causas={payload.causas_raiz_count}")

    # ── PHASE 4: Output (One-Pager + A10 evaluation) ─────────────────────────
    async def _phase_output(self) -> None:
        """Run A8 to produce the One-Pager and A10 to evaluate it."""
        logger.info("═══ PHASE 4: Output | run=%s ═══", self.run_id)

        # 1. LOAD STATE
        node = state_manager.read_handoff(self.run_id, expected_phase="output")
        synth_h = SynthesisHandoff(**node["payload"])
        state_manager.update_pipeline_status(self.run_id, "output")
        self.blackboard.set_estado("output")

        # 2. EXECUTE A8
        a8_output = await self.runner.execute("A8", self.run_id)
        self.results["A8"] = a8_output

        # 3. VALIDATE via A10 (with retry-on-fail)
        passed, feedback = await run_onepager_evaluation(
            diagnostico_id=self.run_id,
            onepager_output=a8_output,
            llm_client=self.llm_client,
        )
        if not passed:
            logger.info("[A10] Retrying A8 with One-Pager evaluator feedback...")
            self.blackboard._data.setdefault("onepager_eval_feedback", feedback)
            self.results["A8"] = await self.runner.execute("A8", self.run_id)
            passed, _ = await run_onepager_evaluation(
                diagnostico_id=self.run_id,
                onepager_output=self.results["A8"],
                llm_client=self.llm_client,
            )
            if not passed:
                # Raises OnePagerEscalationError on retry_count >= 2
                await run_onepager_evaluation(
                    diagnostico_id=self.run_id,
                    onepager_output=self.results["A8"],
                    llm_client=self.llm_client,
                )

        # 4. WRITE HANDOFF (terminal — payload kept for the audit log)
        onepager_path = f"blackboard/onepager/{self.run_id}_A8.json"
        state_manager.update_onepager(
            self.run_id, {"status": "approved" if passed else "generated",
                          "output_path": onepager_path},
        )
        payload = OutputHandoff(
            onepager_path=onepager_path,
            eval_passed=passed,
            delivered=passed,
        )
        state_manager.write_handoff(
            self.run_id,
            current_phase="output",
            next_phase="terminal",
            payload=asdict(payload),
            produced_by="phase_output",
        )

        # 5. RELEASE CONTEXT
        if passed:
            state_manager.update_pipeline_status(self.run_id, "delivered")
        state_manager.append_history(self.run_id, "A8", "complete",
                                     f"delivered={passed}")
```

```python
async def run_agent(
    agent_id: str,
    run_id: str,
    blackboard: Blackboard,
    llm_client: AsyncLLMClient,
) -> dict[str, Any]:
    """
    Backward-compatible shim around AgentRunner.execute().

    Preserves the OTel tracing and SSE event emission so existing callers
    (api/app.py, tests) see no behavioral change.
    """
    from orchestrator.agent_runner_v2 import AgentRunner

    tracer = get_tracer()
    with tracer.start_as_current_span(
        "diagnosticore.agent.run",
        attributes={"agent.id": agent_id, "run.id": run_id},
    ) as span:
        logger.info("─── Agent START | id=%s | run=%s ───", agent_id, run_id)
        event_bus.emit(run_id, "agent_start", {"agent_id": agent_id})
        try:
            runner = AgentRunner(blackboard, llm_client)
            validated_output = await runner.execute(agent_id, run_id)
            nivel = validated_output.get("nivel_madurez")
            if nivel is not None:
                span.set_attribute("agent.nivel_madurez", nivel)
            event_bus.emit(run_id, "agent_done",
                           {"agent_id": agent_id, "nivel_madurez": nivel})
            logger.info("─── Agent DONE  | id=%s | run=%s ───", agent_id, run_id)
            return validated_output
        except Exception as exc:
            span.record_exception(exc)
            span.set_status(Status(StatusCode.ERROR, str(exc)[:120]))
            event_bus.emit(run_id, "agent_error",
                           {"agent_id": agent_id, "error": str(exc)})
            raise
```
